var chat = document.getElementById("chat"); var msgBox = document.getElementById("msg"); var statusEl = document.getElementById("status"); var hist = []; var SYS = "You are JEFE, an AI assistant. You write complete working code."; function log(txt, cls) { var d = document.createElement("div"); d.className = cls; d.textContent = txt; chat.appendChild(d); chat.scrollTop = chat.scrollHeight; } window.doSend = function() { var t = msgBox.value.trim(); if (!t) return; msgBox.value = ""; log("YOU: " + t, "u"); statusEl.textContent = "THINKING..."; statusEl.style.color = "#ff3333"; hist.push({"role": "user", "content": t}); var body = JSON.stringify({ "messages": [{"role": "system", "content": SYS}].concat(hist), "model": "openai" }); fetch("https://text.pollinations.ai/", { method: "POST", headers: {"Content-Type": "application/json"}, body: body }) .then(r => r.text()) .then(txt => { log("JEFE: " + txt, "b"); hist.push({"role": "assistant", "content": txt}); statusEl.textContent = "READY"; statusEl.style.color = "#00ff41"; }) .catch(err => { log("ERROR: " + err, "e"); statusEl.textContent = "ERROR"; }); };
